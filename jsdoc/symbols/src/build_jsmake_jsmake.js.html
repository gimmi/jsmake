 21 /** @class Various helper methods to make working with Javascript easier */
 22 jsmake.Utils = {
 23 	/**
 24 	 * Return the same string with escaped regex chars, in order to be safely included as part of regex
 25 	 * @param {String} str string to escape
 26 	 * @returns {String} escaped string
 27 	 */
 28 	escapeForRegex: function (str) {
 29 		return str.replace(/[\-\[\]{}()*+?.,\\\^$|#\s]/g, "\\$&");
 30 	},
 31 	/**
 32 	 * @param v
 33 	 * @returns {Boolean} true if passed value is an array
 34 	 */
 35 	isArray: function (v) {
 36 		// Check ignored test 'isArray should show strange behavior on Firefox'
 37 		return Object.prototype.toString.apply(v) === '[object Array]';
 38 	},
 39 	/**
 40 	 * @param v
 41 	 * @returns {Boolean} true if passed value is an argument object
 42 	 */
 43 	isArguments: function (v) {
 44 		return !!(v && Object.prototype.hasOwnProperty.call(v, 'callee'));
 45 	},
 46 	/**
 47 	 * @param v
 48 	 * @returns {Array} passed value, converted to array
 49 	 */
 50 	toArray: function (v) {
 51 		if (this.isEmpty(v)) {
 52 			return [];
 53 		} else if (this.isArray(v)) {
 54 			return v;
 55 		} else if (this.isArguments(v)) {
 56 			return Array.prototype.slice.call(v);
 57 		} else {
 58 			return [ v ];
 59 		}
 60 	},
 61 	/**
 62 	 * @param v
 63 	 * @returns {Boolean} true if passed value is an object
 64 	 */
 65 	isObject : function (v) {
 66 		return !!v && Object.prototype.toString.call(v) === '[object Object]';
 67 	},
 68 	/**
 69 	 * @param v
 70 	 * @returns {Boolean} true if passed value is a number
 71 	 */
 72 	isNumber: function (v) {
 73 		return typeof v === 'number' && isFinite(v);
 74 	},
 75 	/**
 76 	 * @param v
 77 	 * @returns {Boolean} true if passed value is null, undefined or empty array
 78 	 */
 79 	isEmpty : function (v) {
 80 		return v === null || v === undefined || ((this.isArray(v) && !v.length));
 81 	},
 82 	/**
 83 	 * @param {String} str string to trim
 84 	 * @returns {String} passed value with head and tail spaces removed
 85 	 */
 86 	trim: function (str) {
 87 		return str.replace(/(?:^\s+)|(?:\s+$)/g, '');
 88 	},
 89 	/**
 90 	 * Iterate over each element of items.
 91 	 * @param items the collection on which iterate, can be anything
 92 	 * @param {Function} fn the funcion to call for each element in items.
 93 	 * Will be called with the following parameters: currentItem, itemIndex, items.
 94 	 * If function returns truthy value then iteration will stop
 95 	 * @param {Object} [scope] 'this' binding for function
 96 	 * @example
 97 	 * // Array iteration: the following code logs
 98 	 * // item=a, index=0, items=[a,b]
 99 	 * // item=b, index=1, items=[a,b]
100 	 * jsmake.Utils.each([ 'a', 'b'], function (item, index, items) {
101 	 *     jsmake.Sys.log('item=' + item + ', index=' + index + ', items=' + items);
102 	 * }, this);
103 	 * // Object iteration: the following code logs
104 	 * // item=1, index=a, items=[object]
105 	 * // item=2, index=b, items=[object]
106 	 * jsmake.Utils.each({ a: 1, b: 2 }, function (item, index, items) {
107 	 *     jsmake.Sys.log('item=' + item + ', index=' + index + ', items=' + items);
108 	 * }, this);
109 	 */
110 	each: function (items, fn, scope) {
111 		var key;
112 		if (this.isObject(items)) {
113 			for (key in items) {
114 				if (items.hasOwnProperty(key)) {
115 					if (fn.call(scope, items[key], key, items)) {
116 						return;
117 					}
118 				}
119 			}
120 		} else {
121 			items = this.toArray(items);
122 			for (key = 0; key < items.length; key += 1) {
123 				if (fn.call(scope, items[key], key, items)) {
124 					return;
125 				}
126 			}
127 		}
128 	},
129 	/**
130 	 * Filter collection, returning elements that satisfy passed criteria
131 	 * @param items can be anything, see {@link jsmake.Utils.each}
132 	 * @param {Function} fn filter criteria, will be called for each element in items, passing current element as parameter.
133 	 * Must return falsy value to indicate that the element should be filtered out
134 	 * @param {Object} [scope] 'this' binding for function
135 	 * @returns {Array} filtered values
136 	 * @example
137 	 * // returns [ 1, 2 ]
138 	 * jsmake.Utils.filter([ 1, 2, 3 ], function (item) {
139 	 *     return item < 3;
140 	 * });
141 	 */
142 	filter: function (items, fn, scope) {
143 		var ret = [];
144 		this.each(items, function (item) {
145 			if (fn.call(scope, item)) {
146 				ret.push(item);
147 			}
148 		}, this);
149 		return ret;
150 	},
151 	/**
152 	 * Transform each item in passed collection, returning a new array with transformed items
153 	 * @param items can be anything, see {@link jsmake.Utils.each}
154 	 * @param {Function} fn transformation function, will be called for each element in items.
155 	 * Will be called with the following parameters: currentItem, itemIndex, items.
156 	 * If function returns truthy value then iteration will stop
157 	 * Must return the transformed item
158 	 * @param {Object} [scope] 'this' binding for function
159 	 * @returns {Array} new array with transformed items
160 	 * @example
161 	 * // returns [ 4, 9 ]
162 	 * jsmake.Utils.map([ 2, 3 ], function (item) {
163 	 *     return item * item;
164 	 * });
165 	 */
166 	map: function (items, fn, scope) {
167 		var ret = [];
168 		this.each(items, function (item, key) {
169 			ret.push(fn.call(scope, item, key, items));
170 		}, this);
171 		return ret;
172 	},
173 	/**
174 	 * @example
175 	 * // returns 'items are: 2 3 '
176 	 * jsmake.Utils.reduce([ 2, 3 ], function (memo, item) {
177 	 *     return memo + item + ' ';
178 	 * }, 'items are: ');
179 	 */
180 	reduce: function (items, fn, memo, scope) {
181 		this.each(items, function (item) {
182 			memo = fn.call(scope, memo, item);
183 		}, this);
184 		return memo;
185 	},
186 	/**
187 	 * @example
188 	 * jsmake.Utils.contains([ 2, 3 ], 3); // returns true
189 	 * jsmake.Utils.contains([ 2, 3 ], 4); // returns false
190 	 */
191 	contains: function (items, item) {
192 		var ret = false;
193 		this.each(items, function (it) {
194 			ret = (it === item);
195 			return ret;
196 		}, this);
197 		return ret;
198 	},
199 	/**
200 	 * @example
201 	 * jsmake.Utils.distinct([ 2, 3, 2, 3 ]); // returns [ 2, 3 ]
202 	 */
203 	distinct: function (items) {
204 		var ret = [];
205 		this.each(items, function (item) {
206 			if (!this.contains(ret, item)) {
207 				ret.push(item);
208 			}
209 		}, this);
210 		return ret;
211 	},
212 	/**
213 	 * @example
214 	 * jsmake.Utils.flatten([ 1, [ 2, 3 ], [ 4, [ 5, 6 ] ] ]); // returns [ 1, 2, 3, 4, 5, 6 ]
215 	 */
216 	flatten: function (items) {
217 		return this.reduce(items, function (memo, item) {
218 			if (this.isArray(item)) {
219 				memo = memo.concat(this.flatten(item));
220 			} else {
221 				memo.push(item);
222 			}
223 			return memo;
224 		}, [], this);
225 	}
226 };

228 jsmake.Project = function (name, defaultTaskName, body, logger) {
229 	this._name = name;
230 	this._defaultTaskName = defaultTaskName;
231 	this._tasks = {};
232 	this._body = body;
233 	this._logger = logger;
234 };
235 jsmake.Project.prototype = {
236 	getName: function () {
237 		return this._name;
238 	},
239 	addTask: function (task) {
240 		this._tasks[task.getName()] = task;
241 	},
242 	getTask: function (name) {
243 		var task = this._tasks[name];
244 		if (!task) {
245 			throw "Task '" + name + "' not defined";
246 		}
247 		return task;
248 	},
249 	getTasks: function (name) {
250 		var tasks = [];
251 		this._fillDependencies(this.getTask(name), tasks, new jsmake.RecursionChecker('Task recursion found'));
252 		return jsmake.Utils.distinct(tasks);
253 	},
254 	runBody: function (global) {
255 		var me = this;
256 		global.task = function (name, tasks, body) {
257 			me.addTask(new jsmake.Task(name, tasks, body, me._logger));
258 		};
259 		this._body.apply({}, []);
260 		global.task = undefined;
261 	},
262 	runTask: function (name, args) {
263 		var tasks, taskNames;
264 		name = name || this._defaultTaskName;
265 		tasks = this.getTasks(name);
266 		taskNames = jsmake.Utils.map(tasks, function (task) {
267 			return task.getName();
268 		}, this);
269 		this._logger.log('Task execution order: ' + taskNames.join(', '));
270 		jsmake.Utils.each(tasks, function (task) {
271 			task.run(task.getName() === name ? args : []);
272 		}, this);
273 	},
274 	_fillDependencies: function (task, tasks, recursionChecker) {
275 		recursionChecker.wrap(task.getName(), function () {
276 			jsmake.Utils.each(task.getTaskNames(), function (taskName) {
277 				var task = this.getTask(taskName);
278 				this._fillDependencies(task, tasks, recursionChecker);
279 			}, this);
280 			tasks.push(task);
281 		}, this);
282 	}
283 };

304 jsmake.RecursionChecker = function (message) {
305 	this._message = message;
306 	this._stack = [];
307 };
308 jsmake.RecursionChecker.prototype = {
309 	enter: function (id) {
310 		this._check(id);
311 		this._stack.push(id);
312 	},
313 	exit: function () {
314 		this._stack.pop();
315 	},
316 	wrap: function (id, fn, scope) {
317 		this.enter(id);
318 		try {
319 			fn.call(scope);
320 		} finally {
321 			this.exit();
322 		}
323 	},
324 	_check: function (id) {
325 		if (jsmake.Utils.contains(this._stack, id)) {
326 			this._stack.push(id);
327 			throw this._message + ': ' + this._stack.join(' => ');
328 		}
329 	}
330 };

332 jsmake.AntPathMatcher = function (pattern, caseSensitive) {
333 	this._pattern = pattern;
334 	this._caseSensitive = caseSensitive;
335 };
336 jsmake.AntPathMatcher.prototype = {
337 	match: function (path) {
338 		var patternTokens, pathTokens;
339 		patternTokens = this._tokenize(this._pattern);
340 		pathTokens = this._tokenize(path);
341 		return this._matchTokens(patternTokens, pathTokens);
342 	},
343 	_matchTokens: function (patternTokens, pathTokens) {
344 		var patternToken, pathToken;
345 		while (true) {
346 			patternToken = patternTokens.shift();
347 			if (patternToken === '**') {
348 				pathTokens = pathTokens.slice(-patternTokens.length).reverse();
349 				patternTokens = patternTokens.reverse();
350 				return this._matchTokens(patternTokens, pathTokens);
351 			}
352 			pathToken = pathTokens.shift();
353 			if (patternToken && pathToken) {
354 				if (!this._matchToken(patternToken, pathToken)) {
355 					return false;
356 				}
357 			} else if (patternToken && !pathToken) {
358 				return false;
359 			} else if (!patternToken && pathToken) {
360 				return false;
361 			} else {
362 				return true;
363 			}
364 		}
365 	},
366 	_matchToken: function (patternToken, pathToken) {
367 		var regex = '', i, ch;
368 		for (i = 0; i < patternToken.length; i += 1) {
369 			ch = patternToken.charAt(i);
370 			if (ch === '*') {
371 				regex += '.*';
372 			} else if (ch === '?') {
373 				regex += '.{1}';
374 			} else {
375 				regex += jsmake.Utils.escapeForRegex(ch);
376 			}
377 		}
378 		return new RegExp('^' + regex + '$', (this._caseSensitive ? '' : 'i')).test(pathToken);
379 	},
380 	_tokenize: function (pattern) {
381 		var tokens = pattern.split(/\\+|\/+/);
382 		tokens = jsmake.Utils.map(tokens, function (token) {
383 			return jsmake.Utils.trim(token);
384 		}, this);
385 		tokens = jsmake.Utils.filter(tokens, function (token) {
386 			return !/^[\s\.]*$/.test(token);
387 		}, this);
388 		if (tokens[tokens.length - 1] === '**') {
389 			throw 'Invalid ** wildcard at end pattern, use **/* instead'; // TODO maybe useless
390 		}
391 		// TODO invalid more then one **
392 		return tokens;
393 	}
394 };

443 /** @namespace Contains methods for working with filesystem */
444 jsmake.Fs = {
445 	/**
446 	 * Create a zip file containing specified file/directory
447 	 * @param {String} srcPath file/directory to zip
448 	 * @param {String} destFile zip file name
449 	 */
450 	zipPath: function (srcPath, destFile) {
451 		jsmake.PathZipper.zip(srcPath, destFile);
452 	},
453 	/**
454 	 * Create a filesystem scanner
455 	 * @param {String} basePath the path to scan for children tha match criteria
456 	 * @returns {jsmake.FsScanner} FsScanner instance to fluently configure and run scanner
457 	 * @see jsmake.FsScanner
458 	 * @example
459 	 * // returns all js and java files in \home folder, including subfolders, excluding .git folders
460 	 * jsmake.Fs.createScanner('\home')
461 	 *     .include('**\*.js')
462 	 *     .include('**\*.java')
463 	 *     .exclude('**\.git')
464 	 *     .scan();
465 	 */
466 	createScanner: function (basePath) {
467 		return new jsmake.FsScanner(basePath, this.isCaseSensitive());
468 	},
469 	/**
470 	 * Return default OS character encoding
471 	 * @returns {String} Character encoding, e.g. 'UTF-8' or 'Cp1252'
472 	 */
473 	getCharacterEncoding: function () {
474 		return java.lang.System.getProperty("file.encoding", "UTF-8"); // Windows default is "Cp1252"
475 	},
476 	/**
477 	 * Return OS path separator
478 	 * @returns {String} path separator, e.g. '/' or '\'
479 	 */
480 	getPathSeparator: function () {
481 		return java.io.File.separator;
482 	},
483 	/**
484 	 * Returns true if OS has case sensitive filesystem
485 	 * @returns {Boolean} true if OS has case sensitive filesystem
486 	 */
487 	isCaseSensitive: function () {
488 		return !jsmake.Sys.isWindowsOs();
489 	},
490 	/**
491 	 * Read text file content
492 	 * @param {String} path path of the file to read
493 	 * @param {String} [characterEncoding=OS default]
494 	 * @returns {String} text content
495 	 */
496 	readFile: function (path, characterEncoding) {
497 		characterEncoding = characterEncoding || this.getCharacterEncoding();
498 		if (!this.fileExists(path)) {
499 			throw "File '" + path + "' not found";
500 		}
501 		return readFile(path, characterEncoding);
502 	},
503 	/**
504 	 * Write String to file, creating all necessary parent directories and overwriting if file already exists
505 	 * @param {String} path path of the file to write
506 	 * @param {String} content file content
507 	 * @param {String} [characterEncoding=OS default]
508 	 */
509 	writeFile: function (path, content, characterEncoding) {
510 		characterEncoding = characterEncoding || this.getCharacterEncoding();
511 		this.createDirectory(this.getParentDirectory(path));
512 		var out = new java.io.FileOutputStream(new java.io.File(path));
513 		content = new java.lang.String(content || '');
514 		try {
515 			out.write(content.getBytes(characterEncoding));
516 		} finally {
517 			out.close();
518 		}
519 	},
520 	/**
521 	 * Extract last element from a path
522 	 * @param {String} path the source path
523 	 * @returns {String} the name of the last element in the path
524 	 * @example
525 	 * jsmake.Fs.getName('/users/gimmi/file.txt'); // returns 'file.txt'
526 	 */
527 	getName: function (path) {
528 		return this._translateJavaString(new java.io.File(path).getName());
529 	},
530 	/**
531 	 * Copy file or directory to another directory
532 	 * @param {String} srcPath source file/directory. Must exists
533 	 * @param {String} destDirectory destination directory
534 	 */
535 	copyPath: function (srcPath, destDirectory) {
536 		if (this.fileExists(srcPath)) {
537 			this._copyFile(srcPath, destDirectory);
538 		} else if (this.directoryExists(srcPath)) {
539 			this._copyDirectory(srcPath, destDirectory);
540 		} else {
541 			throw "Cannot copy source path '" + srcPath + "', it does not exists";
542 		}
543 	},
544 	/**
545 	 * @param {String} path file or directory path
546 	 * @returns {Boolean} true if file or directory exists
547 	 */
548 	pathExists: function (path) {
549 		return new java.io.File(path).exists();
550 	},
551 	/**
552 	 * @param {String} path directory path
553 	 * @returns {Boolean} true if path exists and is a directory
554 	 */
555 	directoryExists: function (path) {
556 		var file = new java.io.File(path);
557 		return file.exists() && file.isDirectory();
558 	},
559 	/**
560 	 * @param {String} path file path
561 	 * @returns {Boolean} true if path exists and is a file
562 	 */
563 	fileExists: function (path) {
564 		var file = new java.io.File(path);
565 		return file.exists() && file.isFile();
566 	},
567 	/**
568 	 * Create directory and all necessary parents
569 	 * @param {String} path directory to create
570 	 */
571 	createDirectory: function (path) {
572 		var file = new java.io.File(path);
573 		if (file.exists() && file.isDirectory()) {
574 			return;
575 		}
576 		if (!file.mkdirs()) {
577 			throw "Failed to create directories for path '" + path + "'";
578 		}
579 	},
580 	/**
581 	 * Delete file or directory, with all cild elements
582 	 * @param {String} path to delete
583 	 */
584 	deletePath: function (path) {
585 		if (!this.pathExists(path)) {
586 			return;
587 		}
588 		jsmake.Utils.each(jsmake.Fs.getChildPathNames(path), function (name) {
589 			this.deletePath(this.combinePaths(path, name));
590 		}, this);
591 		if (!new java.io.File(path)['delete']()) {
592 			throw "'Unable to delete path '" + path + "'";
593 		}
594 	},
595 	/**
596 	 * Transform a path to absolute, removing '.' and '..' references
597 	 * @param {String} path path to translate
598 	 * @returns {String} path in canonical form
599 	 * @example
600 	 * jsmake.Fs.getCanonicalPath('../file.txt'); // returns '/users/file.txt'
601 	 */
602 	getCanonicalPath: function (path) {
603 		return this._translateJavaString(new java.io.File(path).getCanonicalPath());
604 	},
605 	/**
606 	 * Returns parent path
607 	 * @param {String} path
608 	 * @returns {String} parent path
609 	 */
610 	getParentDirectory: function (path) {
611 		return this._translateJavaString(new java.io.File(path).getCanonicalFile().getParent());
612 	},
613 	/**
614 	 * Combine all passed path fragments into one, using OS path separator. Supports any number of parameters.
615 	 * @example
616 	 * jsmake.Fs.combinePaths('home', 'gimmi', [ 'dir/subdir', 'file.txt' ]);
617 	 * // returns 'home/gimmi/dir/subdir/file.txt'
618 	 */
619 	combinePaths: function () {
620 		var paths = jsmake.Utils.flatten(arguments);
621 		return jsmake.Utils.reduce(paths, function (memo, path) {
622 			return (memo ? this._javaCombine(memo, path) : path);
623 		}, null, this);
624 	},
625 	getChildPathNames: function (basePath) {
626 		return this._listFilesWithFilter(basePath, function () {
627 			return true;
628 		});
629 	},
630 	getChildFileNames: function (basePath) {
631 		return this._listFilesWithFilter(basePath, function (fileName) {
632 			return new java.io.File(fileName).isFile();
633 		});
634 	},
635 	getChildDirectoryNames: function (basePath) {
636 		return this._listFilesWithFilter(basePath, function (fileName) {
637 			return new java.io.File(fileName).isDirectory();
638 		});
639 	},
640 	_javaCombine: function (path1, path2) {
641 		return this._translateJavaString(new java.io.File(path1, path2).getPath());
642 	},
643 	_copyDirectory: function (srcDirectory, destDirectory) {
644 		this.deletePath(destDirectory);
645 		this.createDirectory(destDirectory);
646 		jsmake.Utils.each(this.getChildFileNames(srcDirectory), function (path) {
647 			this.copyPath(this.combinePaths(srcDirectory, path), destDirectory);
648 		}, this);
649 		jsmake.Utils.each(this.getChildDirectoryNames(srcDirectory), function (path) {
650 			this.copyPath(this.combinePaths(srcDirectory, path), this.combinePaths(destDirectory, path));
651 		}, this);
652 	},
653 	_copyFile: function (srcFile, destDirectory) {
654 		var destFile = this.combinePaths(destDirectory, this.getName(srcFile));
655 		this.deletePath(destFile);
656 		this.createDirectory(destDirectory);
657 		this._copyFileToFile(srcFile, destFile);
658 	},
659 	_copyFileToFile: function (srcFile, destFile) {
660 		var input, output, buffer, n;
661 		input = new java.io.FileInputStream(srcFile);
662 		try {
663 			output = new java.io.FileOutputStream(destFile);
664 			try {
665 				buffer = java.lang.reflect.Array.newInstance(java.lang.Byte.TYPE, 1024 * 4);
666 				while (-1 !== (n = input.read(buffer))) {
667 					output.write(buffer, 0, n);
668 				}
669 			} finally {
670 				output.close();
671 			}
672 		} finally {
673 			input.close();
674 		}
675 	},
676 	_listFilesWithFilter: function (basePath, filter) {
677 		var fileFilter, files;
678 		fileFilter = new java.io.FileFilter({ accept: filter });
679 		files = this._translateJavaArray(new java.io.File(basePath).listFiles(fileFilter));
680 		return jsmake.Utils.map(files, function (file) {
681 			return this._translateJavaString(file.getName());
682 		}, this);
683 	},
684 	_translateJavaArray: function (javaArray) {
685 		var ary = [], i;
686 		if (javaArray === null) {
687 			return null;
688 		}
689 		for (i = 0; i < javaArray.length; i += 1) {
690 			ary.push(javaArray[i]);
691 		}
692 		return ary;
693 	},
694 	_translateJavaString: function (javaString) {
695 		return String(javaString);
696 	}
697 };
698 /**
699  * Don't instantiate it directly, use {@link jsmake.Fs.createScanner}
700  * @constructor
701  */
702 jsmake.FsScanner = function (basePath, caseSensitive) {
703 	this._basePath = basePath;
704 	this._includeMatchers = [];
705 	this._excludeMatchers = [];
706 	this._caseSensitive = caseSensitive;
707 };
708 jsmake.FsScanner.prototype = {
709 	/**
710 	 * Add a criteria for path inclusion. If no inclusion path are specified, '**\*' is assumed
711 	 * @param {String} pattern
712 	 * @returns {jsmake.FsScanner} this instance, for chaining calls
713 	 * @example
714 	 * jsmake.Fs.createScanner('\home').include('**\*.js').scan();
715 	 */
716 	include: function (pattern) {
717 		this._includeMatchers.push(new jsmake.AntPathMatcher(pattern, this._caseSensitive));
718 		return this;
719 	},
720 	/**
721 	 * Add a criteria for path exclusion
722 	 * @param {String} pattern
723 	 * @returns {jsmake.FsScanner} this instance, for chaining calls
724 	 * @example
725 	 * jsmake.Fs.createScanner('\home').exclude('**\.git').scan();
726 	 */
727 	exclude: function (pattern) {
728 		this._excludeMatchers.push(new jsmake.AntPathMatcher(pattern, this._caseSensitive));
729 		return this;
730 	},
731 	/**
732 	 * Execute filesystem scanning with defined criterias
733 	 * @returns {String[]} all mathing paths
734 	 * @example
735 	 * // returns the path of all files in /home directory
736 	 * jsmake.Fs.createScanner('/home').scan();
737 	 */
738 	scan: function () {
739 		var fileNames = [];
740 		if (this._includeMatchers.length === 0) {
741 			this.include('**/*');
742 		}
743 		this._scan('.', fileNames);
744 		return fileNames;
745 	},
746 	_scan: function (relativePath, fileNames) {
747 		var fullPath = jsmake.Fs.combinePaths(this._basePath, relativePath);
748 		jsmake.Utils.each(jsmake.Fs.getChildFileNames(fullPath), function (fileName) {
749 			fileName = jsmake.Fs.combinePaths(relativePath, fileName);
750 			if (this._evaluatePath(fileName, false)) {
751 				fileNames.push(jsmake.Fs.combinePaths(this._basePath, fileName));
752 			}
753 		}, this);
754 		jsmake.Utils.each(jsmake.Fs.getChildDirectoryNames(fullPath), function (dir) {
755 			dir = jsmake.Fs.combinePaths(relativePath, dir);
756 			if (this._evaluatePath(dir, true)) {
757 				this._scan(dir, fileNames);
758 			}
759 		}, this);
760 	},
761 	_evaluatePath: function (path, def) {
762 		if (this._runMatchers(this._excludeMatchers, path)) {
763 			return false;
764 		}
765 		if (this._runMatchers(this._includeMatchers, path)) {
766 			return true;
767 		}
768 		return def;
769 	},
770 	_runMatchers: function (matchers, value) {
771 		var match = false;
772 		jsmake.Utils.each(matchers, function (matcher) {
773 			match = match || matcher.match(value);
774 		}, this);
775 		return match;
776 	}
777 };

810 jsmake.PathZipper = {
811 	zip: function (srcPath, destFile) {
812 		var zipOutputStream = new java.util.zip.ZipOutputStream(new java.io.FileOutputStream(destFile));
813 		try {
814 			this._zip(jsmake.Fs.getParentDirectory(srcPath), jsmake.Fs.getName(srcPath), zipOutputStream);
815 		} finally {
816 			zipOutputStream.close(); // This raise exception "java.util.zip.ZipException: ZIP file must have at least one entry"
817 		}
818 	},
819 	_zip: function (basePath, relativePath, zipOutputStream) {
820 		var names, path;
821 		path = jsmake.Fs.combinePaths(basePath, relativePath);
822 		if (jsmake.Fs.fileExists(path)) {
823 			this._addFile(basePath, relativePath, zipOutputStream);
824 		} else if (jsmake.Fs.directoryExists(path)) {
825 			jsmake.Utils.each(jsmake.Fs.getChildPathNames(path), function (name) {
826 				this._zip(basePath, jsmake.Fs.combinePaths(relativePath, name), zipOutputStream);
827 			}, this);
828 		} else {
829 			throw "Cannot zip source path '" + path + "', it does not exists";
830 		}
831 	},
832 	_addFile: function (basePath, relativePath, zipOutputStream) {
833 		var fileInputStream, buffer, n;
834 		zipOutputStream.putNextEntry(new java.util.zip.ZipEntry(relativePath));
835 		buffer = java.lang.reflect.Array.newInstance(java.lang.Byte.TYPE, 1024 * 4);
836 		fileInputStream = new java.io.FileInputStream(jsmake.Fs.combinePaths(basePath, relativePath));
837 		try {
838 			while (-1 !== (n = fileInputStream.read(buffer))) {
839 				zipOutputStream.write(buffer, 0, n);
840 			}
841 		} finally {
842 			fileInputStream.close();
843 		}
844 		zipOutputStream.closeEntry();
845 	}
